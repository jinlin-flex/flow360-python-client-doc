[docs]    @classmethod
    def validate(cls, vector):
        """validator for vector"""
        _ = _PydanticValidate(c=vector)
        if not isinstance(vector, cls):
            vector = cls(vector)
        if vector == (0, 0, 0):
            raise pd.ValidationError(ValidationError(f"{cls.__name__} cannot be (0, 0, 0)"), cls)
        return vector
